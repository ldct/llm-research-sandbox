RxJava vs Reactor vs Coroutines

Side-by-side comparison of reactive programming patterns

RxJava 3.1.8
Reactor 3.6.0
Coroutines 1.8.0
RxJava Reactive Streams
Reactor Reactive Streams
Flow Kotlin Functional
Sequential Kotlin Imperative
1

Basic Stream / Iteration

RxJava 
Observable.just("Hello", "World")
    .subscribe(System.out::println);
Reactor 
Flux.just("Hello", "World")
    .subscribe(System.out::println);
Coroutines (Flow) 
flowOf("Hello", "World")
    .collect { println(it) }
Coroutines (Sequential) 
flow {
    flowOf("Hello", "World").collect { emit(it) }
}.collect { println(it) }

Output

Hello
World

FlatMap (Flatten Nested Streams)

RxJava 
Observable.just("Hi", "Rx")
    .flatMap(word -> 
        Observable.fromArray(word.split("")))
    .subscribe(System.out::println);
Reactor 
Flux.just("Hi", "Rx")
    .flatMap(word -> 
        Flux.fromArray(word.split("")))
    .subscribe(System.out::println);
Coroutines (Flow) 
flowOf("Hi", "Rx")
    .flatMapConcat { word -> 
        word.toList().asFlow()
            .map { it.toString() } 
    }.collect { println(it) }
Coroutines (Sequential) 
flow {
    flowOf("Hi", "Rx").collect { word ->
        for (char in word) {
            emit(char.toString())
        }
    }
}.collect { println(it) }

Output

i

Async Operations

RxJava 
Observable.just(1, 2, 3)
    .flatMap(id -> fetchUserAsync(id))
    .subscribe(System.out::println);

// Helper
static Observable<String> fetchUserAsync(int id) {
    return Observable.just("User-" + id)
        .delay(50, TimeUnit.MILLISECONDS);
}
Reactor 
Flux.just(1, 2, 3)
    .flatMap(id -> fetchUserAsync(id))
    .subscribe(System.out::println);

// Helper
static Mono<String> fetchUserAsync(int id) {
    return Mono.just("User-" + id)
        .delayElement(Duration.ofMillis(50));
}
Coroutines (Flow) 
flowOf(1, 2, 3)
    .flatMapConcat { fetchUserFlow(it) }
    .collect { println(it) }

// Helper
fun fetchUserFlow(id: Int) = flow {
    delay(50)
    emit("User-$id")
}
Coroutines (Sequential) 
flow {
    flowOf(1, 2, 3).collect { id ->
        delay(50)
        emit("User-$id")
    }
}.collect { println(it) }

Output

User-1
User-2
User-3

Zip (Combine Streams)

RxJava 
Observable<String> names = 
    Observable.just("Alice", "Bob");
Observable<Integer> ages = 
    Observable.just(25, 30);

Observable.zip(names, ages, 
    (name, age) -> name + " is " + age)
    .subscribe(System.out::println);
Reactor 
Flux<String> names = 
    Flux.just("Alice", "Bob");
Flux<Integer> ages = 
    Flux.just(25, 30);

Flux.zip(names, ages, 
    (name, age) -> name + " is " + age)
    .subscribe(System.out::println);
Coroutines (Flow) 
val names = flowOf("Alice", "Bob")
val ages = flowOf(25, 30)

names.zip(ages) { name, age -> 
    "$name is $age" 
}.collect { println(it) }
Coroutines (Sequential) 
flow {
    val names = flowOf("Alice", "Bob").toList()
    val ages = flowOf(25, 30).toList()
    names.forEachIndexed { i, name ->
        emit("$name is ${ages[i]}")
    }
}.collect { println(it) }

Output

Alice is 25
Bob is 30

Error Handling

RxJava 
Observable.just(1, 2, 0, 4)
    .map(n -> 10 / n)
    .onErrorReturnItem(-1)
    .subscribe(System.out::println);
Reactor 
Flux.just(1, 2, 0, 4)
    .map(n -> 10 / n)
    .onErrorReturn(-1)
    .subscribe(System.out::println);
Coroutines (Flow) 
flowOf(1, 2, 0, 4)
    .map { 10 / it }
    .catch { emit(-1) }
    .collect { println(it) }
Coroutines (Sequential) 
flow {
    flowOf(1, 2, 0, 4).collect { n ->
        emit(10 / n)
    }
}.catch { emit(-1) }
 .collect { println(it) }

Output

10
5
-1

Parallel Execution

RxJava 
Observable<String> a = Observable.just("Data-A")
    .delay(100, TimeUnit.MILLISECONDS);
Observable<String> b = Observable.just("Data-B")
    .delay(50, TimeUnit.MILLISECONDS);

Observable.zip(a, b, (x, y) -> x + ", " + y)
    .subscribe(System.out::println);
Reactor 
Mono<String> a = Mono.just("Data-A")
    .delayElement(Duration.ofMillis(100));
Mono<String> b = Mono.just("Data-B")
    .delayElement(Duration.ofMillis(50));

Mono.zip(a, b, (x, y) -> x + ", " + y)
    .subscribe(System.out::println);
Coroutines (Flow) 
flow {
    coroutineScope {
        val a = async { fetchData("A", 100) }
        val b = async { fetchData("B", 50) }
        emit("${a.await()}, ${b.await()}")
    }
}.collect { println(it) }
Coroutines (Sequential) 
flow {
    coroutineScope {
        val a = async { fetchData("A", 100) }
        val b = async { fetchData("B", 50) }
        emit("${a.await()}, ${b.await()}")
    }
}.collect { println(it) }

Output

Data-A, Data-B

Built with RxJava 3, Project Reactor 3.6 & Kotlin Coroutines 1.8
